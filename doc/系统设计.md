## 一、架构总览

### 1.1 设计哲学

| 维度 | 设计决策 |
|------|----------|
| **审计范围** | 仅当前会话（对话结束即清理） |
| **审计强度** | 会话内不可抵赖（Merkle + HMAC/Ed25519） |
| **LLM 处理** | LLM 输出（Markdown/结论）作为**不可变事件**固化，回退时重放事件恢复，**不重新调用 LLM** |
| **状态一致性** | 先写 SQLite，后更新内存；崩溃时从 SQLite 重建 |
| **元数据传输** | 前后端通过 **JSON 复合结构** 传输，Markdown 仅用于人类阅读/导出 |
| **并发控制** | SQLite 单写者队列（`asyncio.Queue`），杜绝锁竞争 |
| **清理可靠性** | 独立持久化调度表 + 启动补偿扫描，不依赖 `asyncio.sleep` |

### 2.1 会话上下文管理器（Session Context Manager）

**职责**：生命周期管理、隔离、清理调度。

```python
class SessionContextManager:
    def __init__(self):
        self.active_sessions: Dict[str, SessionContext] = {}
        self.cleanup_db = sqlite3.connect("/tmp/synapse_cleanup.db")
        self._init_cleanup_schema()
        self._start_cleanup_loop()
        self._startup_recover()  # 启动时补偿清理
    
    def create_session(self, user_id: str) -> str:
        session_id = f"sess_{user_id}_{uuid4().hex[:8]}_{int(time.time())}"
        ctx = SessionContext(
            session_id=session_id,
            event_store=SessionEventStore(session_id),
            projection=SessionStateProjection(session_id),
            lineage=SessionLineageGraph(session_id),
            rollback=SessionRollbackController(session_id),
            write_queue=asyncio.Queue(),  # 单写者队列
            writer_task=asyncio.create_task(self._writer_loop(session_id))
        )
        self.active_sessions[session_id] = ctx
        return session_id
    
    async def _writer_loop(self, session_id: str):
        """SQLite 单写者队列：杜绝并发锁竞争"""
        ctx = self.active_sessions[session_id]
        while True:
            event = await ctx.write_queue.get()
            if event is None:  #  poison pill
                break
            try:
                ctx.event_store._do_insert(event)  # 同步写入
            except Exception as e:
                logger.error(f"Write failed: {e}")
                # 重试一次，再失败则标记会话损坏
            ctx.write_queue.task_done()
    
    def _init_cleanup_schema(self):
        self.cleanup_db.execute("""
            CREATE TABLE IF NOT EXISTS cleanup_queue (
                session_id TEXT PRIMARY KEY,
                purge_after TEXT NOT NULL,
                status TEXT DEFAULT 'pending',
                created_at TEXT DEFAULT CURRENT_TIMESTAMP
            )
        """)
    
    def schedule_cleanup(self, session_id: str, delay_seconds: int = 300):
        purge_time = (datetime.now() + timedelta(seconds=delay_seconds)).isoformat()
        self.cleanup_db.execute(
            "INSERT OR REPLACE INTO cleanup_queue (session_id, purge_after, status) VALUES (?, ?, 'pending')",
            (session_id, purge_time)
        )
        self.cleanup_db.commit()
    
    async def _cleanup_loop(self):
        """每分钟扫描，不依赖 asyncio.sleep 的可靠性"""
        while True:
            due = self.cleanup_db.execute(
                "SELECT session_id FROM cleanup_queue WHERE status='pending' AND purge_after <= ?",
                (datetime.now().isoformat(),)
            ).fetchall()
            for (sid,) in due:
                await self._purge_session(sid)
            await asyncio.sleep(60)
    
    def _startup_recover(self):
        """启动时补偿清理：处理上次崩溃遗留"""
        overdue = self.cleanup_db.execute(
            "SELECT session_id FROM cleanup_queue WHERE status='pending' AND purge_after < ?",
            (datetime.now().isoformat(),)
        ).fetchall()
        for (sid,) in overdue:
            asyncio.create_task(self._purge_session(sid))
    
    async def _purge_session(self, session_id: str):
        ctx = self.active_sessions.pop(session_id, None)
        if ctx:
            ctx.writer_task.cancel()  # 停止单写者队列
            try:
                await ctx.writer_task
            except asyncio.CancelledError:
                pass
            ctx.event_store.close_and_purge()
            ctx.lineage.purge()
            ctx.projection.purge()
        # 清理快照文件
        for f in glob(f"/tmp/snapshots/{session_id}_*"):
            os.remove(f)
        # 标记完成
        self.cleanup_db.execute(
            "UPDATE cleanup_queue SET status='done' WHERE session_id=?",
            (session_id,)
        )
        self.cleanup_db.commit()
```

### 2.2 会话级事件存储（Session Event Store）

**职责**：仅追加写入、单写者消费、崩溃恢复。

```python
class SessionEventStore:
    def __init__(self, session_id: str):
        self.db_path = f"/tmp/synapse_sessions/{session_id}.db"
        os.makedirs(os.path.dirname(self.db_path), exist_ok=True)
        self.conn = sqlite3.connect(self.db_path, check_same_thread=False)
        self._init_schema()
        self._init_wal()
    
    def _init_schema(self):
        self.conn.executescript("""
            CREATE TABLE IF NOT EXISTS events (
                event_id TEXT PRIMARY KEY,
                event_type TEXT NOT NULL,
                aggregate_id TEXT NOT NULL,
                version INTEGER NOT NULL,
                timestamp TEXT NOT NULL,
                actor TEXT,
                payload TEXT,
                metadata TEXT,
                previous_hash TEXT,
                event_hash TEXT,
                signature TEXT,
                is_llm_output INTEGER DEFAULT 0,
                cost_usd REAL DEFAULT 0,
                tokens_input INTEGER DEFAULT 0,
                tokens_output INTEGER DEFAULT 0,
                model_version TEXT
            );
            CREATE INDEX IF NOT EXISTS idx_events_version ON events(version);
            
            CREATE TABLE IF NOT EXISTS snapshots (
                snapshot_id TEXT PRIMARY KEY,
                version INTEGER NOT NULL,
                state_blob BLOB,
                merkle_root TEXT,
                created_at TEXT DEFAULT CURRENT_TIMESTAMP
            );
            
            PRAGMA journal_mode=WAL;
            PRAGMA busy_timeout=5000;
        """)
    
    def _do_insert(self, event: Event):
        """同步写入，由单写者队列调用"""
        self.conn.execute("""
            INSERT INTO events (event_id, event_type, aggregate_id, version, timestamp,
                              actor, payload, metadata, previous_hash, event_hash,
                              signature, is_llm_output, cost_usd, tokens_input, tokens_output, model_version)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            event.event_id, event.event_type, event.aggregate_id, event.version,
            event.timestamp, json.dumps(event.actor), json.dumps(event.payload),
            json.dumps(event.metadata), event.previous_hash, event.event_hash,
            event.signature, 1 if event.is_llm_output else 0,
            event.cost_usd, event.tokens_input, event.tokens_output, event.model_version
        ))
        self.conn.commit()
    
    def get_events(self, from_version: int = 0, to_version: Optional[int] = None) -> List[Event]:
        cursor = self.conn.execute(
            "SELECT * FROM events WHERE version >= ? ORDER BY version",
            (from_version,)
        )
        rows = cursor.fetchall()
        if to_version:
            rows = rows[:to_version - from_version]
        return [self._deserialize(row) for row in rows]
    
    def get_latest_version(self) -> int:
        row = self.conn.execute("SELECT MAX(version) FROM events").fetchone()
        return row[0] or 0
    
    def save_snapshot(self, version: int, state_blob: bytes, merkle_root: str):
        self.conn.execute(
            "INSERT INTO snapshots (snapshot_id, version, state_blob, merkle_root) VALUES (?, ?, ?, ?)",
            (f"snap_{version}", version, state_blob, merkle_root)
        )
        self.conn.commit()
    
    def get_latest_snapshot(self) -> Optional[Snapshot]:
        row = self.conn.execute(
            "SELECT * FROM snapshots ORDER BY version DESC LIMIT 1"
        ).fetchone()
        return self._deserialize_snapshot(row) if row else None
    
    def close_and_purge(self):
        self.conn.close()
        try:
            os.remove(self.db_path)
        except PermissionError:
            # Windows 句柄未释放，加入重试队列
            asyncio.create_task(self._retry_remove(self.db_path))
    
    async def _retry_remove(self, path: str, retries: int = 5):
        for _ in range(retries):
            await asyncio.sleep(1)
            try:
                os.remove(path)
                return
            except PermissionError:
                continue
        logger.error(f"Failed to remove {path} after {retries} retries")
```

### 2.3 状态投影器（State Projection）

**职责**：先持久化后投影、快照加速、崩溃重建。

```python
class SessionStateProjection:
    SNAPSHOT_INTERVAL = 50  # 每 50 个事件打快照
    
    def __init__(self, session_id: str, event_store: SessionEventStore):
        self.session_id = session_id
        self.event_store = event_store
        self.current_state = State()
        self.merkle_tree = MerkleTree()
        self.current_version = 0
        self._state_dirty = False
    
    async def apply_event(self, event: Event, write_queue: asyncio.Queue):
        # 1. 先提交到写入队列（持久化）
        await write_queue.put(event)
        await write_queue.join()  # 等待写入完成
        
        # 2. 仅在写入成功后更新内存
        try:
            self._project_to_memory(event)
            self.merkle_tree.append(event.event_hash)
            self.current_version = event.version
            self._state_dirty = False
            
            # 3. 定期打快照
            if event.version % self.SNAPSHOT_INTERVAL == 0:
                self._create_snapshot()
                
        except Exception as e:
            logger.error(f"Projection failed: {e}")
            self._mark_state_dirty()
            # 触发从 SQLite 重建
            self._rebuild_from_store()
    
    def _project_to_memory(self, event: Event):
        if event.event_type == "FindingReported":
            self.current_state.findings[event.payload["finding_id"]] = Finding(**event.payload)
        elif event.event_type == "FindingRolledBack":
            fid = event.payload["target_finding_id"]
            if fid in self.current_state.findings:
                self.current_state.findings[fid].status = "rolled_back"
        elif event.event_type == "SynthesisGenerated":
            # LLM 输出固化事件：直接存储生成的 Markdown
            self.current_state.deliverable = Deliverable(
                markdown_content=event.payload["markdown_content"],
                audit_metadata=event.payload["audit_metadata"],
                merkle_root=event.metadata.get("merkle_root")
            )
        # ... 其他事件类型
    
    def _create_snapshot(self):
        state_blob = pickle.dumps(self.current_state)
        self.event_store.save_snapshot(
            version=self.current_version,
            state_blob=state_blob,
            merkle_root=self.merkle_tree.root_hash
        )
    
    def get_state_at(self, target_version: int) -> State:
        # 1. 找最近的快照
        snapshot = self.event_store.get_latest_snapshot()
        if snapshot and snapshot.version <= target_version:
            state = pickle.loads(snapshot.state_blob)
            start_version = snapshot.version
        else:
            state = State()
            start_version = 0
        
        # 2. 仅重放快照之后的事件
        events = self.event_store.get_events(from_version=start_version)
        for event in events:
            if event.version > target_version:
                break
            self._apply_to_state(state, event)
        return state
    
    def _rebuild_from_store(self):
        """崩溃恢复：从 SQLite 全量重建"""
        self.current_state = State()
        self.merkle_tree = MerkleTree()
        events = self.event_store.get_events()
        for event in events:
            self._project_to_memory(event)
            self.merkle_tree.append(event.event_hash)
        self.current_version = len(events)
        self._state_dirty = False
    
    def _mark_state_dirty(self):
        self._state_dirty = True
    
    def purge(self):
        self.current_state = None
        self.merkle_tree = None
```

### 2.4 血缘引擎（Session Lineage Engine）

**职责**：内存图 + 按需序列化进 SQLite（防多进程丢失）。

```python
class SessionLineageGraph:
    def __init__(self, session_id: str, event_store: SessionEventStore):
        self.session_id = session_id
        self.event_store = event_store
        self.graph = nx.DiGraph()
        self._load_from_events()  # 启动时从 SQLite 重建
    
    def _load_from_events(self):
        """从 SQLite 事件重建图（防崩溃后内存丢失）"""
        events = self.event_store.get_events()
        for event in events:
            self._apply_event_to_graph(event)
    
    def add_finding(self, finding: Finding, source_doc: SourceDocument):
        self.graph.add_node(finding.finding_id, type="Finding", **finding)
        self.graph.add_node(source_doc.doc_id, type="SourceDocument", **source_doc)
        self.graph.add_edge(finding.finding_id, source_doc.doc_id, relation="EXTRACTED_FROM")
        self._persist_graph()  # 序列化到 SQLite（防多进程丢失）
    
    def _persist_graph(self):
        """将图序列化为 JSON 存入 SQLite session_state 表"""
        graph_data = nx.node_link_data(self.graph)
        self.event_store.conn.execute("""
            CREATE TABLE IF NOT EXISTS session_state (key TEXT PRIMARY KEY, value TEXT)
        """)
        self.event_store.conn.execute(
            "INSERT OR REPLACE INTO session_state (key, value) VALUES (?, ?)",
            ("lineage_graph", json.dumps(graph_data))
        )
        self.event_store.conn.commit()
    
    def _apply_event_to_graph(self, event: Event):
        if event.event_type == "FindingReported":
            # 重建节点和边
            self.graph.add_node(event.payload["finding_id"], type="Finding")
            self.graph.add_edge(event.payload["finding_id"], event.payload["source"], relation="EXTRACTED_FROM")
        elif event.event_type == "ConflictDetected":
            self.graph.add_edge(event.payload["finding_a"], event.payload["finding_b"], relation="CONFLICTS_WITH")
        # ... 其他事件
    
    def trace_upstream(self, node_id: str) -> List[str]:
        return list(nx.ancestors(self.graph, node_id))
    
    def trace_downstream(self, node_id: str) -> List[str]:
        return list(nx.descendants(self.graph, node_id))
    
    def get_impact(self, finding_id: str) -> ImpactReport:
        downstream = self.trace_downstream(finding_id)
        return ImpactReport(
            affected_synthesis=[n for n in downstream if self.graph.nodes[n].get("type") == "Synthesis"],
            affected_deliverables=[n for n in downstream if self.graph.nodes[n].get("type") == "Deliverable"]
        )
    
    def purge(self):
        self.graph.clear()
        del self.graph
```

### 2.5 回退控制器（Session Rollback Controller）

**职责**：补偿事件生成、快照恢复、不重新调用 LLM。

```python
class SessionRollbackController:
    def __init__(self, session_id: str, event_store: SessionEventStore, 
                 projection: SessionStateProjection, lineage: SessionLineageGraph):
        self.session_id = session_id
        self.event_store = event_store
        self.projection = projection
        self.lineage = lineage
    
    async def rollback_finding(self, finding_id: str, reason: str, actor: str,
                               write_queue: asyncio.Queue) -> RollbackResult:
        # 1. 检查范围
        if not self.lineage.graph.has_node(finding_id):
            raise RollbackError("Finding not in current session")
        
        # 2. 计算影响（仅当前会话内）
        impact = self.lineage.get_impact(finding_id)
        
        # 3. 生成补偿事件
        compensation = Event(
            event_type="FindingRolledBack",
            aggregate_id=self.session_id,
            version=self.projection.current_version + 1,
            payload={
                "target_finding_id": finding_id,
                "reason": reason,
                "actor": actor,
                "affected_downstream": impact.affected_synthesis + impact.affected_deliverables
            },
            previous_hash=self.projection.merkle_tree.root_hash
        )
        
        # 4. 追加到事件流（通过单写者队列）
        await self.projection.apply_event(compensation, write_queue)
        
        # 5. 恢复当时的交付物（从事件流重放，不重新调用 LLM）
        # 找到最后一个 SynthesisGenerated 事件，恢复其状态
        last_synthesis = self._find_last_synthesis_event()
        if last_synthesis:
            self.projection.current_state.deliverable = Deliverable(
                markdown_content=last_synthesis.payload["markdown_content"],
                audit_metadata=last_synthesis.payload["audit_metadata"]
            )
        
        return RollbackResult(
            new_version=self.projection.current_version,
            affected_count=len(impact.affected_downstream),
            status="success"
        )
    
    def _find_last_synthesis_event(self) -> Optional[Event]:
        """从 SQLite 查找最近的 SynthesisGenerated 事件"""
        row = self.event_store.conn.execute(
            "SELECT * FROM events WHERE event_type='SynthesisGenerated' ORDER BY version DESC LIMIT 1"
        ).fetchone()
        return self.event_store._deserialize(row) if row else None
    
    async def create_branch(self, at_version: int, write_queue: asyncio.Queue) -> str:
        branch_id = f"{self.session_id}_branch_{uuid4().hex[:8]}"
        # 复制 SQLite 文件
        shutil.copy(self.event_store.db_path, f"/tmp/synapse_sessions/{branch_id}.db")
        return branch_id
```

### 2.6 完整性验证器（Integrity Verifier）

**职责**：Merkle 验证 + 内容签名 + 密钥管理。

```python
class IntegrityVerifier:
    def __init__(self, secret_key: Optional[bytes] = None):
        # MVP: 从环境变量加载 HMAC 密钥
        self.hmac_key = secret_key or os.environ.get("AUDIT_HMAC_SECRET", "").encode()
        if len(self.hmac_key) < 32:
            raise ValueError("HMAC secret must be at least 32 bytes")
    
    def compute_event_hash(self, event: Event) -> str:
        payload = f"{event.event_type}:{event.version}:{event.payload}:{event.previous_hash}"
        return hashlib.sha256(payload.encode()).hexdigest()
    
    def build_merkle_root(self, events: List[Event]) -> str:
        hashes = [self.compute_event_hash(e) for e in events]
        return self._merkle_tree_root(hashes)
    
    def sign_content(self, content: str) -> str:
        """对 Markdown 正文进行 HMAC 签名"""
        return hmac.new(self.hmac_key, content.encode(), hashlib.sha256).hexdigest()
    
    def verify_report(self, markdown_content: str, events: List[Event]) -> VerificationResult:
        # 1. 从 Markdown 头部提取声明的 Merkle 根和内容签名
        declared_merkle = extract_merkle_root(markdown_content)
        declared_content_sig = extract_content_signature(markdown_content)
        
        # 2. 重新计算 Merkle 根
        computed_merkle = self.build_merkle_root(events)
        
        # 3. 重新计算内容签名
        # 提取纯 Markdown 正文（去除元数据头部）
        pure_content = extract_pure_markdown(markdown_content)
        computed_content_sig = self.sign_content(pure_content)
        
        return VerificationResult(
            merkle_valid=hmac.compare_digest(declared_merkle, computed_merkle),
            content_valid=hmac.compare_digest(declared_content_sig, computed_content_sig),
            declared_merkle=declared_merkle,
            computed_merkle=computed_merkle
        )
    
    def _merkle_tree_root(self, hashes: List[str]) -> str:
        if not hashes:
            return hashlib.sha256(b"").hexdigest()
        while len(hashes) > 1:
            next_level = []
            for i in range(0, len(hashes), 2):
                left = hashes[i]
                right = hashes[i+1] if i+1 < len(hashes) else left
                combined = hashlib.sha256(f"{left}{right}".encode()).hexdigest()
                next_level.append(combined)
            hashes = next_level
        return hashes[0]
```

### 2.7 综合 Agent（Synthesis Agent）—— LLM 输出固化

**关键修正**：LLM 生成的 Markdown 作为**不可变事件**写入 SQLite，回退时直接恢复，不重新调用 LLM。

```python
class SynthesisAgent:
    def __init__(self, model: str = "claude-3-sonnet-20240229"):
        self.model = model
        self.client = anthropic.Anthropic()
    
    async def generate(self, state: State, task_spec: TaskSpec) -> Event:
        # 1. 构建提示（含所有发现和冲突）
        prompt = self._build_synthesis_prompt(state, task_spec)
        
        # 2. 调用 LLM
        response = await self.client.messages.create(
            model=self.model,
            max_tokens=4000,
            messages=[{"role": "user", "content": prompt}]
        )
        raw_output = response.content[0].text
        
        # 3. 解析为 Markdown + 元数据
        markdown_content, audit_metadata = self._parse_synthesis_output(raw_output)
        
        # 4. 计算成本
        cost = self._calculate_cost(response.usage, self.model)
        
        # 5. 生成 SynthesisGenerated 事件（固化 LLM 输出）
        return Event(
            event_type="SynthesisGenerated",
            aggregate_id=task_spec.task_id,
            version=0,  # 由 Orchestrator 分配实际版本
            payload={
                "markdown_content": markdown_content,
                "audit_metadata": audit_metadata,
                "prompt_template_version": "synthesis_v1.0"
            },
            metadata={
                "cost_usd": cost,
                "tokens_input": response.usage.input_tokens,
                "tokens_output": response.usage.output_tokens,
                "model_version": self.model
            },
            is_llm_output=True  # 标记为 LLM 生成内容
        )
    
    def _parse_synthesis_output(self, raw: str) -> Tuple[str, dict]:
        # 解析 LLM 输出为 Markdown 正文 + 审计元数据
        # 元数据以 JSON 块形式嵌入在 Markdown 底部
        parts = raw.split("---AUDIT_METADATA---")
        markdown = parts[0].strip()
        metadata = json.loads(parts[1]) if len(parts) > 1 else {}
        return markdown, metadata
```

```
   └── 写入 SQLite

   └── 写入 SQLite

   └── 从 SQLite 读取最新 SynthesisGenerated 事件

   └── 写入 cleanup_schedule 表
   └── 5分钟后清理进程删除文件
```

## 六、最终结论

| 维度 | 修正前风险 | 修正后状态 |
|------|-----------|-----------|
| **LLM 非确定性** | 回退时重新调用 LLM，结果不一致 | **LLM 输出固化事件，回退时重放恢复** |
| **SQLite 并发** | 多 Agent 写入锁竞争 | **单写者队列，零锁竞争** |
| **内存状态孤岛** | 多进程部署丢失状态 | **序列化进 SQLite + 启动重建，或单进程约束** |
| **状态一致性** | 内存与 SQLite 可能分裂 | **先持久化后投影，崩溃自动重建** |
| **元数据解析** | 正则解析 HTML 注释，脆弱 | **JSON 复合传输，Markdown 仅用于人类阅读** |
| **Merkle 验证** | 仅验证事件链，不验证内容 | **内容签名纳入 Merkle，验证 Markdown 正文完整性** |
| **清理可靠性** | `asyncio.sleep` 不可靠 | **持久化调度表 + 启动补偿扫描** |
| **重放性能** | 全量重放 O(n) | **快照加速，从最近快照重放** |

**SynapseGrid 3.1-Final 是一份经过深度修正、可工程落地的完整系统设计。MVP 版本可在 3 个月内交付，满足"可审计、可追溯、可回退"三大核心特性，同时保持存储成本趋近于零。**
